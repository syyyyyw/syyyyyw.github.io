P1002 铺地毯

代码如下：

#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=1e4+5;
int a[maxn],b[maxn],g[maxn],k[maxn],n,x,y;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++) {
        scanf("%d%d%d%d",&a[i],&b[i],&g[i],&k[i]);
    }
    scanf("%d%d",&x,&y);
    int ans=-1;
    for(int i=1;i<=n;i++) {
        int temp1=a[i]+g[i],temp2=b[i]+k[i];
        if(x>=a[i]&&x<=temp1&&y>=b[i]&&y<=temp2) {
            ans=i;
        }
    }
    printf("%d\n",ans);
}

P1067 多项式输出

#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int n,a[10005];
int main()
{
    while(~scanf("%d",&n)) {
        for(int i=n;i>=0;i--) {
            scanf("%d",&a[i]);
        }
        for(int i=n;i>=0;i--) {
            if(a[i]==0) {
                continue;
            }
            if(a[i]>0&&i!=n) {
                printf("+");
            }
            if(a[i]==1||a[i]==-1) {
                if(a[i]==-1) {
                    printf("-");
                }
                if(i==0) {
                    printf("1");
                }
            }
            else{
                printf("%d",a[i]);
            }
            if(i!=0) {
                printf("x");
            }
            if(i!=1&&i!=0) {
                printf("^%d",i);
            }
        }
        printf("\n");
    }
}

P1056 排座椅

#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int m,n,k,l,d;
struct Node
{
    int num,id;
}a[1005],b[1005];
bool cmp(Node a,Node b)
{
    return a.num>b.num;
}
bool cmp2(Node a,Node b)
{
    return a.id<b.id;
}
int main()
{
    scanf("%d%d%d%d%d",&m,&n,&k,&l,&d);
    for(int i=1;i<=m;i++) {
        a[i].num=0;
        a[i].id=i;
    }
    for(int i=1;i<=n;i++) {
        b[i].id=i;
        b[i].num=0;
    }
    for(int i=0;i<d;i++) {
        int x,y,p,q;
        scanf("%d%d%d%d",&x,&y,&p,&q);
        if(x==p) {
            int temp=min(y,q);
            b[temp].num++;
        }
        else{
            int temp=min(x,p);
            a[temp].num++;
        }
    }
    sort(a,a+m,cmp);
    sort(b,b+n,cmp);
    sort(a,a+k,cmp2);
    sort(b,b+l,cmp2);
    for(int i=0;i<k;i++) {
        printf("%d%c",a[i].id,i==k-1?'\n':' ');
    }
    for(int i=0;i<l;i++) {
        printf("%d%c",b[i].id,i==l-1?'\n':' ');
    }
}

P1328 生活大爆炸版石头剪刀布

#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int n1,n2,n3;
int a[205],b[205];
int main()
{
    scanf("%d%d%d",&n1,&n2,&n3);
    for(int i=0;i<n2;i++) {
        scanf("%d",&a[i]);
    }
    for(int i=0;i<n3;i++) {
        scanf("%d",&b[i]);
    }
    int ans1=0,ans2=0;
    for(int i=0;i<n1;i++) {
        int p1=i%n2,p2=i%n3;
        if(a[p1]==0) {
            if(b[p2]==2||b[p2]==3) {
                ans1++;
            }
            else if(b[p2]==1||b[p2]==4) {
                ans2++;
            }
        }
        if(a[p1]==1) {
            if(b[p2]==3||b[p2]==0) {
                ans1++;
            }
            else if(b[p2]==2||b[p2]==4) {
                ans2++;
            }
        }
        if(a[p1]==2) {
            if(b[p2]==4||b[p2]==1) {
                ans1++;
            }
            else if(b[p2]==3||b[p2]==0) {
                ans2++;
            }
        }
        if(a[p1]==3) {
            if(b[p2]==4||b[p2]==2) {
                ans1++;
            }
            else if(b[p2]==0||b[p2]==1) {
                ans2++;
            }
        }
        if(a[p1]==4) {
            if(b[p2]==0||b[p2]==1) {
                ans1++;

                ans2++;

     }
     printf("%d %d\n",ans1,ans2);
}

P1563 玩具谜题

#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=1e5+7;
int n,m;
char s[maxn][15];
int f[maxn],a[maxn],ff[maxn];
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++) {
        scanf("%d%s",&f[i],s[i]);
    }
    for(int i=0;i<m;i++) {
        scanf("%d%d",&a[i],&ff[i]);
    }
    int p=0;
    for(int i=0;i<m;i++) {
        if(a[i]==f[p]) {
            p=(p-ff[i]+n)%n;
        }
        else{
            p=(p+ff[i])%n;
        }
    }
    printf("%s\n",s[p]);
}
